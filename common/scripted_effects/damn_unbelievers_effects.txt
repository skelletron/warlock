damn_unbelievers_effect = {
	every_living_character = {
		if = {
			limit = {
				NOT = {
					faith = faith:great_old_one
				}
			}
			add_to_list = people_to_damn
			# add_character_modifier = { modifier = level_5_vein_snare_victim_modifier years = 10 }
		}
		if = {
			limit = {
				NOT = {
					faith = faith:great_old_one
				}
				is_ruler = yes
				OR = {
					highest_held_title_tier = tier_kingdom
					highest_held_title_tier = tier_empire
				}
			}
			add_to_list = damned_rulers
		}
	}
	every_in_list = {
		list = people_to_damn
		save_scope_as = current_person
		scope:current_person = {
			add_character_modifier = { modifier = level_6_vein_snare_victim_modifier days = 1000 }
		}
	}
}
