namespace = summon_great_old_one

summon_great_old_one.1001 = {
	type = character_event
	title = summon_great_old_one.1001.t
	desc = summon_great_old_one.1001.desc
	theme = default
	left_portrait = owner

	option = {
		name = summon_great_old_one.1001.a
		custom_tooltip = summon_great_old_one.1001.a_1.tt

		hidden_effect = {
			every_living_character = {
				if = {
					limit = {
						NOT = {
							faith = faith:great_old_one
						}
					}
					add_to_list = people_to_damn
					# add_character_modifier = { modifier = level_5_vein_snare_victim_modifier years = 10 }
				}
			}
			every_in_list = {
				list = people_to_damn
				save_scope_as = current_person
				scope:current_person = {
					add_character_modifier = { modifier = level_5_vein_snare_victim_modifier days = 1000 }
				}
			}
		}
	}
}
